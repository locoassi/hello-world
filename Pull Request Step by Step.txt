Pull Request Step by Step

# Clone repository
git clone https://github.com/locoassi/hello-world.git

# Go to the directory
cd hello-world

# Create a new branch
git checkout -b hello-world-test

# Stage and commit the change
git add .
git commit -m "add PR txt"

# Check commit
git status

# Push change to repository
git push origin hello-world-test
-username: locoassi
-credential: accese-token
